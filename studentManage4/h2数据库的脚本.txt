create table USERS (
	id bigint primary key auto_increment, 
	user_name varchar(30) not null, 
	real_name varchar(30) not null, 
	password varchar(50) not null
	);
	
	
create table STUDENT (
	id bigint primary key auto_increment, 
	name varchar(30) not null, 
	age int, 
	email varchar(50)
	);
